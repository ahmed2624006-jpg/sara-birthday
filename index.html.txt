    <script>
        // دالة التأكد من كلمة السر
        function checkPassword() {
            const pass = document.getElementById('passInput').value.toLowerCase().trim();
            if (pass === 'i love you') {
                document.getElementById('password-screen').style.opacity = '0';
                setTimeout(() => {
                    document.getElementById('password-screen').style.display = 'none';
                    document.getElementById('main-content').style.display = 'block';
                    initHearts(); // بدء أنيميشن القلوب
                }, 800);
            } else { 
                alert('كلمة السر غلط يا سوسو! ❤️'); 
            }
        }

        // دالة تغيير المرحلة داخل الظرف (الصور والرسائل)
        function nextStage(envNum) {
            const img = document.getElementById('img' + envNum);
            const txt = document.getElementById('txt' + envNum);
            
            img.style.opacity = '0'; // تأثير اختفاء
            
            setTimeout(() => {
                if (envNum === 1) {
                    img.src = "https://i.postimg.cc/13pQdmQs/Screenshot-2026-05-04-060037.png";
                    txt.innerHTML = "Ever since we met, my heart knew where it wanted to stay—with you, in every soft moment, every smile, every quiet piece of forever.";
                } else {
                    img.src = "https://i.postimg.cc/PNs7bDNF/Screenshot-2026-05-04-060239.png";
                    txt.innerHTML = "Even through screens and pixels, your laugh reaches me like sunlight through a window—warm, real, and impossible to forget.";
                }
                img.style.opacity = '1'; // تأثير ظهور
            }, 500);
            
            event.stopPropagation(); // لمنع إغلاق الظرف تلقائياً
        }

        // دالة إنشاء القلوب المتطايرة
        function initHearts() {
            setInterval(() => {
                const h = document.createElement('div');
                h.innerHTML = '❤️'; h.className = 'heart';
                h.style.left = Math.random() * 100 + 'vw';
                h.style.fontSize = Math.random() * 20 + 15 + 'px';
                document.body.appendChild(h);
                setTimeout(() => h.remove(), 5000);
            }, 300);
        }

        // دالة التحكم في الأغنية
        let playing = false;
        function toggleMusic() {
            const player = document.getElementById('player');
            const btn = document.querySelector('.music-btn');
            if (!playing) {
                player.src += "&autoplay=1";
                btn.innerHTML = "⏸️ إيقاف الأغنية";
                playing = true;
            } else {
                player.src = "https://www.youtube.com/embed/5F-0h_2S7eQ?enablejsapi=1";
                btn.innerHTML = "🎵 تشغيل الأغنية";
                playing = false;
            }
        }
    </script>
